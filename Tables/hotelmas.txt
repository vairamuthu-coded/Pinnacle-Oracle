 CREATE TABLE ASPTBLHOSTELMAS
(
  ASPTBLHOSTELMASID  INTEGER PRIMARY KEY,
  COMPCODE          INTEGER,
  HOSTELNAME    VARCHAR2(50),
  HOSTELBLOCK       VARCHAR2(50),
  HOSTELROOM VARCHAR2(50),
  ACTIVE VARCHAR2(1),
  USERNAME INTEGER,  
  MODIFIED          VARCHAR2(50),
  CREATEDON         VARCHAR2(50),
  IPADDRESS         VARCHAR2(100 BYTE)
)

ALTER TABLE ASPTBLHOSTELMAS ADD    FOREIGN KEY (COMPCODE)   
 REFERENCES TIPLBS.GTCOMPMAST (GTCOMPMASTID);
 
 ALTER TABLE ASPTBLHOSTELMAS ADD    FOREIGN KEY (USERNAME)   
 REFERENCES TIPLBS.asptblusermas (USERID);
 
 CREATE SEQUENCE ASPTBLHOSTELMASSEQ
  START WITH 1
  MAXVALUE 99999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
  NOORDER;
  
 
 DROP TRIGGER ASPTBLHOSTELMASTRI
  
  CREATE OR REPLACE TRIGGER TIPLBS.ASPTBLHOSTELMASTRI
BEFORE INSERT ON TIPLBS.ASPTBLHOSTELMAS REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLHOSTELMASID INTEGER;
BEGIN 
ASPTBLHOSTELMASID:= 0;
   SELECT ASPTBLHOSTELMASSEQ.NEXTVAL INTO ASPTBLHOSTELMASID FROM DUAL;
   :NEW.ASPTBLHOSTELMASID:= ASPTBLHOSTELMASID;
END ASPTBLHOSTELMASTRI;
