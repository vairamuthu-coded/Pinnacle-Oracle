DROP TABLE HOSTELGATEPASS

CREATE TABLE ASPTBLHOSTELGATEPASS
(
  ASPTBLHOSTELGATEPASSID  INTEGER,
  COMPCODE          INTEGER,
  IDCARDNO          INTEGER,
  EMPNAME           INTEGER,
  DEPARTMENT        INTEGER,
  HOSTELNAME        VARCHAR2(50 BYTE),
  HOSTELBLOCK       INTEGER,
  HOSTELROOM        INTEGER,
  CONTACTNO         INTEGER,
  SYSTEMDATE        VARCHAR2(50 BYTE),
  SYSTEMTIME        VARCHAR2(50 BYTE),
  RESERVEDHRS       VARCHAR2(50 BYTE),
  REASON            INTEGER,
  EMPIMAGE          BLOB,
  QRCODE            BLOB,
  INTIME            VARCHAR2(50 BYTE),
  OUTTIME           VARCHAR2(50 BYTE),
  USERNAME          INTEGER,
  MODIFIED          DATE,
  CREATEDON         DATE,
  IPADDRESS         VARCHAR2(100 BYTE),
  IPADDRESS1        INTEGER

ALTER TABLE ASPTBLHOSTELGATEPASS ADD (
  FOREIGN KEY (COMPCODE)   
 REFERENCES GTCOMPMAST (GTCOMPMASTID),
  FOREIGN KEY (DEPARTMENT) 
 REFERENCES ASPTBLDEP (ASPTBLDEPID),
  FOREIGN KEY (EMPNAME) 
 REFERENCES ASPTBLEMP (ASPTBLEMPID));
    
     CREATE SEQUENCE ASPTBLHOSTELGATEPASSSEQ
  START WITH 100001
  MAXVALUE 99999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
  NOORDER;
  
  CREATE OR REPLACE TRIGGER ASPTBLHOSTELGATEPASSTRI
BEFORE INSERT ON ASPTBLHOSTELGATEPASS REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLHOSTELGATEPASSID INTEGER;
BEGIN 
ASPTBLHOSTELGATEPASSID:= 0;
   SELECT ASPTBLHOSTELGATEPASSSEQ.NEXTVAL INTO ASPTBLHOSTELGATEPASSID FROM DUAL;
   :NEW.ASPTBLHOSTELGATEPASSID:= ASPTBLHOSTELGATEPASSID;
END ASPTBLHOSTELGATEPASSTRI;
