DROP TABLE ASPTTMS.TFTDEVICE CASCADE CONSTRAINTS;

CREATE TABLE ASPTTMS.TFTDEVICE
(
  TFTDEVICEID    INTEGER NOT NULL,
  USER_ID        INTEGER,
  FINGER_INDEX   INTEGER,
  FINGER_IMAGE   CLOB,
  PRIVILEGE      INTEGER,
  PASSWORDS      VARCHAR2(50 BYTE),
  FLAG           INTEGER,
  ENABLED        VARCHAR2(10 BYTE),
  NAME           VARCHAR2(50 BYTE),
  MACIP          VARCHAR2(50 BYTE),
  CARDNUMBER     VARCHAR2(50 BYTE),
  FACE_IMAGE     CLOB,
  FACEIMAGE      BLOB,
  FACELENGTH     INTEGER,
  FINGER_LENGTH  INTEGER
)

DROP SEQUENCE ASPTTMS.TFTDEVICESEQ;

CREATE SEQUENCE ASPTTMS.TFTDEVICESEQ
  START WITH 8334
  MAXVALUE 9999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;


CREATE OR REPLACE TRIGGER ASPTTMS.TFTDEVICETRI
BEFORE INSERT
ON ASPTTMS.TFTDevice REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
TFTDeviceID NUMERIC;
BEGIN 
TFTDeviceID := 0;
   SELECT TFTDeviceseq.NEXTVAL INTO TFTDeviceID FROM DUAL;
   :NEW.TFTDeviceID:= TFTDeviceID;
END TFTDevicetri;
