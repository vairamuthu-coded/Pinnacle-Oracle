CREATE TABLE ASPTTMS.gtcompmast
(
  gtcompmastID  INTEGER NOT NULL,
  COMPCODE     VARCHAR2(5 BYTE),
  COMPNAME     VARCHAR2(100 BYTE),
  ADDRESS      VARCHAR2(500 BYTE),
  CITY         INT,
  STATE       INT,
  COUNTRY    INT,
  PINCODE      VARCHAR2(50 BYTE),
  PANNO        VARCHAR2(50 BYTE),
  CSTNO        VARCHAR2(50 BYTE),
  GSTDATE      DATE,
  PHONENO      INTEGER,
  FAXNO        VARCHAR2(50 BYTE),
  EMAIL        VARCHAR2(100 BYTE),
  WEBSITE      VARCHAR2(50 BYTE),
  ACTIVE       VARCHAR2(1 BYTE),
  USERNAME     VARCHAR2(50 BYTE),
  CREATEDON    VARCHAR2(50 BYTE),
  CREATEDBY    VARCHAR2(50 BYTE),
  MODIFIEDON   VARCHAR2(50 BYTE),
  IPADDRESS    VARCHAR2(30 BYTE),
    PRIMARY KEY(gtcompmastID),
        FOREIGN KEY(CITY) REFERENCES TBLCITMAS(TBLCITMASID),
        FOREIGN KEY(STATE) REFERENCES TBLSTAMAS(TBLSTAMASID),
          FOREIGN KEY(COUNTRY) REFERENCES TBLCOUMAS(TBLCOUMASID)
)



CREATE OR REPLACE TRIGGER ASPTTMS.gtcompmasttri
BEFORE INSERT
ON ASPTTMS.gtcompmast REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
gtcompmastID NUMERIC;
BEGIN 
gtcompmastID := 0;
   SELECT gtcompmastSEQ.NEXTVAL INTO gtcompmastID FROM DUAL;
   :NEW.gtcompmastID:= gtcompmastID;
END gtcompmasttri;
/


DROP SEQUENCE ASPTTMS.TBLCITMASSEQ;

CREATE SEQUENCE ASPTTMS.gtcompmastSEQ
  START WITH 1
  MAXVALUE 9999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;
