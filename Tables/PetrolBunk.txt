CREATE TABLE ASPTBLPETMAS(
ASPTBLPETMASID INTEGER,
COMPCODE INTEGER,
BUNKNAME VARCHAR2(100),
BUNKADDRESS VARCHAR2(500),
CONTACTNO  VARCHAR2(50)
)

DROP TABLE ASPTBLPETMAS

CREATE OR REPLACE TRIGGER TIPLAGF.ASPTBLPETMASTRI
BEFORE INSERT
ON TIPLAGF.ASPTBLPETMAS REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLPETMASID INTEGER;
BEGIN 
ASPTBLPETMASID:= 0;
   SELECT ASPTBLPETMASDEQ.NEXTVAL INTO ASPTBLPETMASID FROM DUAL;
   :NEW.ASPTBLPETMASID:= ASPTBLPETMASID;
END ASPTBLPETMASTRI;


DROP SEQUENCE TIPLAGF.ASPTBLPETMASDEQ;

CREATE SEQUENCE TIPLAGF.ASPTBLPETMASDEQ
  START WITH 1
  MAXVALUE 9999
  MINVALUE 1
  NOCYCLE
  NOCACHE
  NOORDER;
