DROP TABLE ASPTTMS.ASPTBLMACIPENTRY CASCADE CONSTRAINTS;

CREATE TABLE ASPTTMS.ASPTBLMACIPENTRY
(
  ASPTBLMACIPENTRYID  INTEGER                   NOT NULL,
  USERID              INTEGER,
  COMPCODE            INTEGER,
  COMPNAME            VARCHAR2(250 BYTE),
  COMPCODE1           VARCHAR2(20 BYTE),
  IPADDRESS           VARCHAR2(30 BYTE),
  LOCATION            VARCHAR2(1000 BYTE),
  MODIFIEDON          DATE,
  CREATEDON           DATE,
  ACTIVE              VARCHAR2(1 BYTE)
)

DROP SEQUENCE ASPTTMS.ASPTBLMACIPENTRYSEQ;

CREATE SEQUENCE ASPTTMS.ASPTBLMACIPENTRYSEQ
  START WITH 3
  MAXVALUE 99999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
  NOORDER;


CREATE OR REPLACE TRIGGER ASPTTMS.ASPTBLMACIPENTRYTRI
BEFORE INSERT
ON ASPTTMS.ASPTBLMACIPENTRY REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLMACIPENTRYID INTEGER;
BEGIN 
ASPTBLMACIPENTRYID:= 0;
   SELECT ASPTBLMACIPENTRYSEQ.NEXTVAL INTO ASPTBLMACIPENTRYID FROM DUAL;
   :NEW.ASPTBLMACIPENTRYID:= ASPTBLMACIPENTRYID;
END ASPTBLMACIPENTRYTRI;
/

-------------------------------------------------------------------------------

DROP TABLE ASPTTMS.ASPTBLMACIPENTRYDET CASCADE CONSTRAINTS;

CREATE TABLE ASPTTMS.ASPTBLMACIPENTRYDET
(
  ASPTBLMACIPENTRYDETID  INTEGER                NOT NULL,
  ASPTBLMACIPENTRYID     INTEGER,
  COMPCODE               VARCHAR2(20 BYTE),
  COMPCODE1              VARCHAR2(20 BYTE),
  MTYPE                  VARCHAR2(10 BYTE),
  MACIP                  VARCHAR2(15 BYTE),
  DUPCHK                 VARCHAR2(50 BYTE),
  MACNO                  NUMBER(10),
  MTYPE2                 VARCHAR2(30 BYTE),
  CURMAC                 VARCHAR2(10 BYTE)
)


ALTER TABLE ASPTTMS.ASPTBLMACIPENTRYDET ADD (
  FOREIGN KEY (ASPTBLMACIPENTRYID) 
  REFERENCES ASPTTMS.ASPTBLMACIPENTRY (ASPTBLMACIPENTRYID)
  ENABLE VALIDATE);
