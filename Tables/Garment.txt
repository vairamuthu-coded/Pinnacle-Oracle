DROP TABLE ASPTTMS.ASPTBLBUYSAM CASCADE CONSTRAINTS;

CREATE TABLE ASPTTMS.ASPTBLBUYSAM
(
  ASPTBLBUYSAMID    INTEGER,
  BUYERCODE     VARCHAR2(10 BYTE),
  STYLEREF      VARCHAR2(50 BYTE),
  STYLENUMBER   VARCHAR2(50 BYTE),
  FABRIC        VARCHAR2(100 BYTE),
  CONTENT       VARCHAR2(50 BYTE),
  COUNTS        VARCHAR2(50 BYTE),
  GAUGE         VARCHAR2(10 BYTE),
  GSM           VARCHAR2(10 BYTE),
  COLOR         VARCHAR2(50 BYTE),
  GARMENTIMAGE  BLOB,
  IMAGEBYTE     INTEGER,
  QRCODE        BLOB,
  ACTIVE        VARCHAR2(1 BYTE),
  RATE          VARCHAR2(50 BYTE)
)


DROP SEQUENCE ASPTTMS.ASPTBLBUYSAMSEQ;

CREATE SEQUENCE ASPTTMS.ASPTBLBUYSAMSEQ
  START WITH 27
  MAXVALUE 9999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;


CREATE OR REPLACE TRIGGER ASPTTMS.ASPTBLBUYSAMTRI
BEFORE INSERT
ON ASPTTMS.ASPTBLBUYSAM REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLBUYSAMID NUMERIC;
BEGIN 
ASPTBLBUYSAMID := 0;
   SELECT ASPTBLBUYSAMseq.NEXTVAL INTO ASPTBLBUYSAMID FROM DUAL;
   :NEW.ASPTBLBUYSAMID:= ASPTBLBUYSAMID;
END ASPTBLBUYSAMtri;
/
