
CREATE TABLE TIPLBS.ASPTBLPARAM
(
  asptblparamid  INTEGER                        NOT NULL,
  COMPCODE       INTEGER,
  USERID         INTEGER,
  APPNAME        VARCHAR2(20 BYTE),
  DSOURCE        VARCHAR2(10 BYTE),
  CSPWORD        VARCHAR2(10 BYTE),
  EPWORD         INTEGER,
  ALIASNAME      VARCHAR2(50 BYTE),
  ACTIVE         VARCHAR2(1 BYTE),
   PRIMARY KEY(asptblparamid),
  FOREIGN KEY(COMPCODE) REFERENCES gtcompmast(gtcompmastID)
)
CREATE OR REPLACE TRIGGER ASPTTMS.ASPTBLPARAMtri
BEFORE INSERT
ON ASPTTMS.ASPTBLPARAM REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
asptblparamid NUMERIC;
BEGIN 
asptblparamid := 0;
   SELECT ASPTBLPARAMSEQ.NEXTVAL INTO asptblparamid FROM DUAL;
   :NEW.asptblparamid:= asptblparamid;
END ASPTBLPARAMtri;
/


CREATE SEQUENCE ASPTTMS.ASPTBLPARAMSEQ
  START WITH 1
  MAXVALUE 9999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
