CREATE TABLE TIPLBS.HRVEHMAST
(
  HRVEHMASTID  INTEGER PRIMARY KEY,
  COMPCODE           INTEGER,
  VEHICLENO             VARCHAR2(50 BYTE),
  VEHICLETYPE  VARCHAR2(50 BYTE),
  USERNAME           INTEGER,
  MODIFIED           VARCHAR2(50 BYTE),
  CREATEDON          VARCHAR2(50 BYTE),
  IPADDRESS          VARCHAR2(100 BYTE),
  ACTIVE             VARCHAR2(1 BYTE),
FOREIGN KEY (COMPCODE)  REFERENCES TIPLBS.GTCOMPMAST (GTCOMPMASTID)
)

CREATE OR REPLACE TRIGGER TIPLBS.HRVEHMASTTRI
BEFORE INSERT ON TIPLBS.HRVEHMAST REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
HRVEHMASTID INTEGER;
BEGIN 
HRVEHMASTID:= 0;
   SELECT HRVEHMASTSEQ.NEXTVAL INTO HRVEHMASTID FROM DUAL;
   :NEW.HRVEHMASTID:= HRVEHMASTID;
END HRVEHMASTTRI;

CREATE SEQUENCE TIPLBS.HRVEHMASTSEQ
  START WITH 1
  MAXVALUE 99999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
  NOORDER;
