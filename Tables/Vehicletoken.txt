CREATE SEQUENCE TIPLBS.ASPTBLVEHTOKENSEQ
  START WITH 1
  MAXVALUE 99999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
  NOORDER;
  
  
CREATE TABLE TIPLBS.ASPTBLVEHTOKEN
(
  ASPTBLVEHTOKENID  INTEGER PRIMARY KEY,
    FINYEAR           INTEGER,
  COMPCODE           INTEGER,  
  TOKENNO             VARCHAR2(50 BYTE),
  TOKENDATE  DATE,
  VEHILENO INTEGER,
  VEHILETYPE INTEGER,
  EMPNAME INTEGER,
  ACTIVE VARCHAR(1),
  TOKENCANCEL VARCHAR2(1),
  TOKENIMAGE BLOB,
  TOTALLITRES INT,
  REMARKS VARCHAR2(500),
  USERNAME           INTEGER,
  MODIFIED           VARCHAR2(50 BYTE),
  CREATEDON          VARCHAR2(50 BYTE),
  IPADDRESS          VARCHAR2(100 BYTE),
  FOREIGN KEY (FINYEAR)  REFERENCES TIPLBS.TBLFINMAS (TBLFINMASID),
FOREIGN KEY (COMPCODE)  REFERENCES TIPLBS.GTCOMPMAST (GTCOMPMASTID),
FOREIGN KEY (USERNAME)  REFERENCES TIPLBS.asptblusermas (USERID)
)




CREATE OR REPLACE TRIGGER TIPLBS.ASPTBLVEHTOKENTRI
BEFORE INSERT ON TIPLBS.ASPTBLVEHTOKEN REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLVEHTOKENID INTEGER;
BEGIN 
ASPTBLVEHTOKENID:= 0;
   SELECT ASPTBLVEHTOKENSEQ.NEXTVAL INTO ASPTBLVEHTOKENID FROM DUAL;
   :NEW.ASPTBLVEHTOKENID:= ASPTBLVEHTOKENID;
END ASPTBLVEHTOKENTRI;


  

  CREATE SEQUENCE TIPLBS.ASPTBLVEHTOKENDETSEQ
  START WITH 1
  MAXVALUE 99999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
  NOORDER;
  
  
  CREATE TABLE TIPLBS.ASPTBLVEHTOKENDET
(
  ASPTBLVEHTOKENDETID  INTEGER PRIMARY KEY,
  ITEMNAME           INTEGER,
  KM           INTEGER,  
  LITRES             VARCHAR2(50 BYTE),
  ASPTBLVEHTOKENID  INTEGER NOT NULL,
NOTES VARCHAR2(100 BYTE),
  FOREIGN KEY (ITEMNAME)  REFERENCES TIPLBS.GTGENITEMMAST (GTGENITEMMASTID),
FOREIGN KEY (ASPTBLVEHTOKENID)  REFERENCES TIPLBS.ASPTBLVEHTOKEN (ASPTBLVEHTOKENID)

ASPTBLVEHTOKENDETID INTEGER;
