CREATE TABLE ASPTBLMENUNAME
(
  MENUNAMEID    INTEGER,
  MENUNAME      VARCHAR2(50 BYTE),
  ACTIVE        VARCHAR2(5 BYTE),
  PARENTMENUID  INTEGER,
  CREATEON      VARCHAR2(50 BYTE),
  CREATEBY      VARCHAR2(50 BYTE),
  MODIFYON      VARCHAR2(50 BYTE),
  IPADDRESS     VARCHAR2(50 BYTE)
)

CREATE OR REPLACE TRIGGER ASPTBLMENUNAMETRI
BEFORE INSERT
ON ASPTBLMENUNAME REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
MENUNAMEID INTEGER;
BEGIN 
MENUNAMEID:= 0;
   SELECT ASPTBLMENUNAMESEQ.NEXTVAL INTO MENUNAMEID FROM DUAL;
   :NEW.MENUNAMEID:= MENUNAMEID;
END ASPTBLMENUNAMETRI;

CREATE SEQUENCE ASPTBLMENUNAMESEQ
  START WITH 1
  MAXVALUE 99999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;
