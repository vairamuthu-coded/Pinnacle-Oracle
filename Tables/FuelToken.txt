CREATE OR REPLACE TRIGGER TIPLAGF.ASPTBLVEHTOKENTRI
BEFORE INSERT ON TIPLAGF.ASPTBLVEHTOKEN REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLVEHTOKENID INTEGER;
BEGIN 
ASPTBLVEHTOKENID:= 0;
   SELECT ASPTBLVEHTOKENSEQ.NEXTVAL INTO ASPTBLVEHTOKENID FROM DUAL;
   :NEW.ASPTBLVEHTOKENID:= ASPTBLVEHTOKENID;
END ASPTBLVEHTOKENTRI;


DROP trigger TIPLAGF.ASPTBLVEHTOKENTRI;

CREATE SEQUENCE TIPLAGF.ASPTBLVEHTOKENSEQ
  START WITH 1
  MAXVALUE 99999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
  NOORDER;
