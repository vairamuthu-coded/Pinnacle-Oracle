  CREATE TABLE TBLCOUMAS
(
  TBLCOUMASID  INTEGER          PRIMARY KEY,
  COUNTRY      VARCHAR2(50 BYTE),
  ACTIVE       VARCHAR2(1 BYTE),
  USERNAME     VARCHAR2(50 BYTE),
  MODIFIEDON   DATE,
  CREATEDBY    VARCHAR2(50 BYTE),
  CREATEDON    DATE,
  IPADD        VARCHAR2(50 BYTE)
)

CREATE OR REPLACE TRIGGER TBLCOUMASTRI
BEFORE INSERT
ON TBLCOUMAS REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
TBLCOUMASID NUMERIC;
BEGIN 
TBLCOUMASID := 0;
   SELECT TBLCOUMASTRISEQ.NEXTVAL INTO TBLCOUMASID FROM DUAL;
   :NEW.TBLCOUMASID:= TBLCOUMASID;
END TBLCOUMASTRI;


CREATE SEQUENCE TBLCOUMASTRISEQ
  START WITH 1
  MAXVALUE 9999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;
