CREATE TABLE TBLSTAMAS
(
  TBLSTAMASID  INTEGER                          NOT NULL,
  STATE        VARCHAR2(100 BYTE),
  COUNTRY      INTEGER,
  ACTIVE       VARCHAR2(1 BYTE),
  USERNAME     VARCHAR2(50 BYTE),
  MODIFIEDON   DATE,
  CREATEDBY    VARCHAR2(50 BYTE),
  CREATEDON    DATE,
  IPADD        VARCHAR2(50 BYTE),
  primary key(TBLSTAMASID),
    FOREIGN KEY(COUNTRY) REFERENCES TBLCOUMAS(TBLCOUMASid)
)


CREATE OR REPLACE TRIGGER TBLSTAMAStri
BEFORE INSERT
ON TBLSTAMAS REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
TBLSTAMASID NUMERIC;
BEGIN 
TBLSTAMASID := 0;
   SELECT TBLSTAMASSEQ.NEXTVAL INTO TBLSTAMASID FROM DUAL;
   :NEW.TBLSTAMASID:= TBLSTAMASID;
END TBLSTAMAStri;
/



CREATE SEQUENCE TBLSTAMASSEQ
  START WITH 1
  MAXVALUE 9999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
