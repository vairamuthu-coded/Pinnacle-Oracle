CREATE TABLE ASPTBLNAVIGATION
(
  MENUID        INTEGER,
  MENUNAME      VARCHAR2(100 BYTE),
  NAVURL        VARCHAR2(200 BYTE),
  PARENTMENUID  INTEGER,
  ACTIVE        VARCHAR2(1 BYTE),
  MENUNAMEID    INTEGER,
  COMPCODE      INTEGER,
  USERNAME      INTEGER,
  FOREIGN KEY(COMPCODE) REFERENCES GTCOMPMAST(GTCOMPMASTID),
   FOREIGN KEY(USERNAME) REFERENCES asptblusermas (USERID)
)

CREATE OR REPLACE TRIGGER ASPTBLNAVIGATIONTRI
BEFORE INSERT ON ASPTBLNAVIGATION REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
MENUID INTEGER;
BEGIN 
MENUID:= 0;
   SELECT ASPTBLNAVIGATIONSEQ.NEXTVAL INTO MENUID FROM DUAL;
   :NEW.MENUID:= MENUID;
END ASPTBLNAVIGATIONTRI;

CREATE SEQUENCE ASPTBLNAVIGATIONSEQ
  START WITH 1
  MAXVALUE 999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;
