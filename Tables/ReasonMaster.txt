CREATE TABLE ASPTBLREASONMAS
(
  ASPTBLREASONMASID  INTEGER PRIMARY KEY,
  COMPCODE           INTEGER,
  REASON             VARCHAR2(50 BYTE),
  USERNAME           INTEGER,
  MODIFIED           VARCHAR2(50 BYTE),
  CREATEDON          VARCHAR2(50 BYTE),
  IPADDRESS          VARCHAR2(100 BYTE),
  ACTIVE             VARCHAR2(1 BYTE)
)

ALTER TABLE ASPTBLREASONMAS ADD    FOREIGN KEY (COMPCODE)   
 REFERENCES GTCOMPMAST (GTCOMPMASTID);
 
 CREATE SEQUENCE ASPTBLREASONMASSEQ
  START WITH 1
  MAXVALUE 99999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
  NOORDER;
  
  CREATE OR REPLACE TRIGGER ASPTBLREASONMASTRI
BEFORE INSERT ON ASPTBLREASONMAS REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLREASONMASID INTEGER;
BEGIN 
ASPTBLREASONMASID:= 0;
   SELECT ASPTBLREASONMASSEQ.NEXTVAL INTO ASPTBLREASONMASID FROM DUAL;
   :NEW.ASPTBLREASONMASID:= ASPTBLREASONMASID;
END ASPTBLREASONMASTRI;
