CREATE TABLE TBLFINMAS
(
  TBLFINMASID  INTEGER prIMARY KEY,
  FINYEAR      VARCHAR2(20 BYTE),
  FROMDATE     VARCHAR2(15 BYTE),
  TODATE       VARCHAR2(15 BYTE),
  CURRENTYEAR  VARCHAR2(1 BYTE),
  TOTALDAYS    INTEGER,
  COMPCODE     INTEGER, 
  USERNAME INTEGER, 
  MODIFIEDON   VARCHAR2(50 BYTE),
  CREATEDON    VARCHAR2(50 BYTE),
  CREAEDBY     VARCHAR2(50 BYTE),
  IPADDRESS    VARCHAR2(50 BYTE),
  FOREIGN KEY(COMPCODE) REFERENCES GTCOMPMAST(GTCOMPMASTID)
)

CREATE OR REPLACE TRIGGER TBLFINMASTRI
BEFORE INSERT 
ON TBLFINMAS REFERENCING NEW AS  NEW OLD AS OLD
FOR EACH ROW
DECLARE
TBLFINMASID INTEGER;
BEGIN
TBLFINMASID :=0;
SELECT TBLFINMASSEQ.NEXTVAL INTO TBLFINMASID FROM DUAL;
:NEW.TBLFINMASID :=TBLFINMASID;
END TBLFINMASTRI;

CREATE SEQUENCE TBLFINMASSEQ
  START WITH 1
  MAXVALUE 9999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;
