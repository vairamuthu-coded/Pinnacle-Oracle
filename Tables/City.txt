CREATE TABLE ASPTTMS.TBLCITMAS
(
  TBLCITMASID  INTEGER                          NOT NULL,
  CITY         VARCHAR2(100 BYTE),
  STATE        INTEGER,
  COUNTRY      INTEGER,
  ACTIVE       VARCHAR2(1 BYTE),
  USERNAME     VARCHAR2(50 BYTE),
  MODIFIEDON   DATE,
  CREATEDBY    VARCHAR2(50 BYTE),
  CREATEDON    DATE,
  IPADD        VARCHAR2(50 BYTE),
  primary key(TBLCITMASID),
       FOREIGN KEY(STATE) REFERENCES TBLSTAMAS(TBLSTAMASid),
          FOREIGN KEY(COUNTRY) REFERENCES TBLCOUMAS(TBLCOUMASid)
)


CREATE OR REPLACE TRIGGER ASPTTMS.TBLCITMAStri
BEFORE INSERT
ON ASPTTMS.TBLCITMAS REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
TBLCITMASID NUMERIC;
BEGIN 
TBLCITMASID := 0;
   SELECT TBLCITMASseq.NEXTVAL INTO TBLCITMASID FROM DUAL;
   :NEW.TBLCITMASID:= TBLCITMASID;
END TBLCITMAStri;
/



CREATE SEQUENCE ASPTTMS.TBLCITMASseq
  START WITH 1
  MAXVALUE 9999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
