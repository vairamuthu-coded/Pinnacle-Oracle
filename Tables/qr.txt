DROP TABLE ASPTBLBUYSAM

CREATE TABLE ASPTBLBUYSAM
(
  ASPTBLBUYSAMID    INTEGER,
  BUYERCODE     VARCHAR2(10 BYTE),
  STYLEREF      VARCHAR2(50 BYTE),
  STYLENUMBER   VARCHAR2(50 BYTE),
  FABRIC        VARCHAR2(100 BYTE),
  CONTENT       VARCHAR2(50 BYTE),
  COUNTS        VARCHAR2(50 BYTE),
  GAUGE         VARCHAR2(10 BYTE),
  GSM           VARCHAR2(10 BYTE),
  COLOR         VARCHAR2(50 BYTE),
  GARMENTIMAGE  BLOB,
  IMAGEBYTE     INTEGER,
  QRCODE        BLOB,
  ACTIVE        VARCHAR2(1 BYTE),
  RATE          VARCHAR2(50 BYTE),
  GIMAGE        CLOB,
  SIZENAME      VARCHAR2(25 BYTE)
)

CREATE OR REPLACE TRIGGER ASPTBLBUYSAMtri
BEFORE INSERT
ON ASPTBLBUYSAM REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLBUYSAMID NUMERIC;
BEGIN 
ASPTBLBUYSAMID := 0;
   SELECT ASPTBLBUYSAMseq.NEXTVAL INTO ASPTBLBUYSAMID FROM DUAL;
   :NEW.ASPTBLBUYSAMID:= ASPTBLBUYSAMID;
END ASPTBLBUYSAMtri;
/



CREATE SEQUENCE ASPTBLBUYSAMseq
  START WITH 1
  MAXVALUE 9999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;
