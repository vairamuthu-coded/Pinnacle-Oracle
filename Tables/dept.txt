
CREATE TABLE ASPTBLDEP
(
  ASPTBLDEPID  INTEGER PRIMARY KEY NOT NULL,
  DEPARTMENT   VARCHAR2(50 BYTE),
  ACTIVE       VARCHAR2(3 BYTE),
  COMPCODE     INTEGER,
  IPADDRESS    VARCHAR2(50 BYTE),
  CREATEDON    VARCHAR2(50 BYTE),
  CREATEDBY    VARCHAR2(50 BYTE),
  MODIFIEDON   VARCHAR2(50 BYTE),
  FOREIGN KEY (COMPCODE) 
 REFERENCES GTCOMPMAST (GTCOMPMASTID)
)



CREATE OR REPLACE TRIGGER ASPTBLDEPTRI
BEFORE INSERT ON ASPTBLDEP REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLDEPID INTEGER;
BEGIN 
ASPTBLDEPID:= 0;
   SELECT ASPTBLDEPSEQ.NEXTVAL INTO ASPTBLDEPID FROM DUAL;
   :NEW.ASPTBLDEPID:= ASPTBLDEPID;
END ASPTBLDEPTRI;



CREATE SEQUENCE ASPTBLDEPSEQ
  START WITH 1
  MAXVALUE 9999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;
