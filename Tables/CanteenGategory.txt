CREATE TABLE TIPLAGF.ASPTBLCANCATEGORYMAS
(
  ASPTBLCANCATEGORYMASID  INTEGER PRIMARY KEY,
  CATEGORY                VARCHAR2(100 BYTE),
  CATEGORYIMAGE           BLOB,
  ACTIVE                  VARCHAR2(1 BYTE),
  USERNAME                INTEGER,
  MODIFIED                VARCHAR2(50 BYTE),
  CREATEDON               VARCHAR2(50 BYTE),
  IPADDRESS               VARCHAR2(50 BYTE)
)

CREATE OR REPLACE TRIGGER ASPTBLCANCATEGORYMASTRI
BEFORE INSERT
ON ASPTBLCANCATEGORYMAS REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLCANCATEGORYMASID NUMERIC;
BEGIN 
ASPTBLCANCATEGORYMASID := 0;
   SELECT ASPTBLCANCATEGORYMASSEQ.NEXTVAL INTO ASPTBLCANCATEGORYMASID FROM DUAL;
   :NEW.ASPTBLCANCATEGORYMASID:= ASPTBLCANCATEGORYMASID;
END ASPTBLCANCATEGORYMASTRI;
/

drop sequence ASPTBLCANCATEGORYMASSEQ


CREATE SEQUENCE ASPTBLCANCATEGORYMASSEQ
  START WITH 1
  MAXVALUE 9999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
