CREATE TABLE TIPLPGC.TRS_TEMPATTLOG
(
  ATTLOGID        NUMBER(20)                    NOT NULL,
  MACHINENUMBER   NUMBER(3)                     NOT NULL,
  IPADDRESS       VARCHAR2(50 BYTE)             NOT NULL,
  ENROLLNO        NUMBER(10)                    NOT NULL,
  VERIFYMODE      NUMBER(2)                     NOT NULL,
  INOUTMODE       NUMBER(2)                     NOT NULL,
  WORKCODE        NUMBER(2)                     NOT NULL,
  DATETIMERECORD  VARCHAR2(50 BYTE)             NOT NULL
)


CREATE OR REPLACE TRIGGER TRS_TEMPATTLOGTRI
BEFORE INSERT ON TRS_TEMPATTLOG REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ATTLOGID INTEGER;
BEGIN 
ATTLOGID:= 0;
   SELECT TRS_TEMPATTLOGSEQ.NEXTVAL INTO ATTLOGID FROM DUAL;
   :NEW.ATTLOGID:= ATTLOGID;
END TRS_TEMPATTLOGTRI;


CREATE SEQUENCE TRS_TEMPATTLOGSEQ
  START WITH 1
  MAXVALUE 999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;
