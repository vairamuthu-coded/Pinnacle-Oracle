CREATE TABLE TIPLAGF.ASPTBLFUELRATEMAS
(
  ASPTBLFUELRATEMASID  INTEGER,
  FINYEAR            INTEGER,
  COMPCODE           INTEGER,  
  FUELDATE          DATE,
  USERNAME           INTEGER,
  MODIFIED           VARCHAR2(50 BYTE),
  CREATEDON          VARCHAR2(50 BYTE),
  IPADDRESS          VARCHAR2(100 BYTE) 
)

CREATE OR REPLACE TRIGGER TIPLAGF.ASPTBLFUELRATEMASTRI
BEFORE INSERT ON TIPLAGF.ASPTBLFUELRATEMAS REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLFUELRATEMASID INTEGER;
BEGIN 
ASPTBLFUELRATEMASID:= 0;
   SELECT ASPTBLFUELRATEMASSEQ.NEXTVAL INTO ASPTBLFUELRATEMASID FROM DUAL;
   :NEW.ASPTBLFUELRATEMASID:= ASPTBLFUELRATEMASID;
END ASPTBLFUELRATEMASTRI;

CREATE SEQUENCE TIPLAGF.ASPTBLFUELRATEMASSEQ
  START WITH 1
  MAXVALUE 99999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
  NOORDER;
  
  CREATE TABLE TIPLAGF.ASPTBLFUELRATEMASDET
(
  ASPTBLFUELRATEMASDETID  INTEGER,
  ASPTBLFUELRATEMASID  INTEGER, 
  COMPCODE           INTEGER,  
   ITEMNAME            INTEGER,
  FUELRATE          DECIMAL
)

CREATE OR REPLACE TRIGGER TIPLAGF.ASPTBLFUELRATEMASDETTRI
BEFORE INSERT ON TIPLAGF.ASPTBLFUELRATEMASDET REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLFUELRATEMASDETID INTEGER;
BEGIN 
ASPTBLFUELRATEMASDETID:= 0;
   SELECT ASPTBLFUELRATEMASDETSEQ.NEXTVAL INTO ASPTBLFUELRATEMASDETID FROM DUAL;
   :NEW.ASPTBLFUELRATEMASDETID:= ASPTBLFUELRATEMASDETID;
END ASPTBLFUELRATEMASDETTRI;

CREATE SEQUENCE TIPLAGF.ASPTBLFUELRATEMASDETSEQ
  START WITH 1
  MAXVALUE 99999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
  NOORDER;
