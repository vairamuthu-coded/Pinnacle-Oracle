CREATE TABLE  ASPTBLCANITEMMAS
(
  ASPTBLCANITEMMASID  INTEGER                          NOT NULL,
  ITEMCODE         VARCHAR2(100 BYTE), 
   CANITEMNAME         VARCHAR2(100 BYTE), 
    CATEGORY INT,
   ITEMCOST DECIMAL,
   ITEMIMAGE BLOB,
  ACTIVE       VARCHAR2(1 BYTE),
  USERNAME     INT,
  MODIFIED   VARCHAR2(50 BYTE),
  CREATEDON    VARCHAR2(50 BYTE),
  IPADDRESS        VARCHAR2(50 BYTE),
   FOREIGN KEY (CATEGORY)   REFERENCES ASPTBLCANCATEGORYMAS(ASPTBLCANCATEGORYMASID),
     FOREIGN KEY (USERNAME)   REFERENCES asptblusermas(USERID)
)


CREATE OR REPLACE TRIGGER ASPTBLCANITEMMASTRI
BEFORE INSERT
ON ASPTBLCANITEMMAS REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
ASPTBLCANITEMMASID NUMERIC;
BEGIN 
ASPTBLCANITEMMASID := 0;
   SELECT ASPTBLCANITEMMASseq.NEXTVAL INTO ASPTBLCANITEMMASID FROM DUAL;
   :NEW.ASPTBLCANITEMMASID:= ASPTBLCANITEMMASID;
END ASPTBLCANITEMMASTRI;
/



CREATE SEQUENCE ASPTBLCANITEMMASseq
  START WITH 1
  MAXVALUE 9999999
  MINVALUE 1
  NOCYCLE
  NOCACHE
